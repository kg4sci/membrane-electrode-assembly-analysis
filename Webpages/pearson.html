'''
1
material type: E
Cu containing: YES
first product: C2H4
MEA configuration: AEM
anolyte: KHCO3
membrane: Sustainion
anode: IrOx
'''
import pandas as pd
import seaborn as sns
import numpy as np
import matplotlib.pyplot as plt
#
df=pd.read_excel('C:\\Users\\hp\\Desktop\\keshe\\ex1.xlsx')
# first_product
df = df.drop('first_product', axis=1)
df.rename(columns={'First product faraday efficiency': 'First product Faradaic efficiency'}, inplace=True)
# CO2 flow rate CO flow rate
data = df.rename(columns={'CO2 flow rate': 'CO flow rate'})

# 
corr_matrix = data.corr(method='pearson')
mask = np.triu(np.ones_like(corr_matrix, dtype=bool))
np.fill_diagonal(mask, False)
# 
plt.figure(figsize=(10, 8))
sns.heatmap(corr_matrix, mask=mask, annot=True, cmap='coolwarm', fmt=".2f", vmin=-1, vmax=1, linewidths=0.5)
plt.title('Pearson Correlation Coefficient Heatmap')
# plt.xticks(rotation=45, ha='right', rotation_mode='anchor')
plt.savefig('fenlei1.jpg', dpi=1000, bbox_inches = 'tight')
plt.show()
